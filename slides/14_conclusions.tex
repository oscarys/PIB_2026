\begin{frame}{Conclusions \& Future Directions}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Summary}
      \begin{itemize}
        \item qEEG has matured into a rich, multi-method discipline
        \item Automated preprocessing (ASR, ICLabel) has largely solved
          artifact rejection
        \item Connectivity and source imaging provide brain-network perspectives
        \item Deep learning achieves state-of-the-art on classification benchmarks
        \item Foundation models (LaBraM) suggest a paradigm shift ahead
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{Open frontiers}
      \begin{itemize}
        \item \textbf{Multimodal}: simultaneous EEG-fMRI-fNIRS
        \item \textbf{Wearable / ambulatory} HD-EEG (dry electrodes, OPM-MEG)
        \item \textbf{Causal biomarkers}: TMS-EEG perturbational approaches
        \item \textbf{LLMs + EEG}: decoding imagined speech
        \item \textbf{Federated learning} for privacy-preserving multi-site models
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \begin{center}
    \alert{\textit{%
      ``The brain is a noisy, non-stationary, high-dimensional system
      — and that is exactly what makes it interesting.''}}
  \end{center}
\end{frame}
