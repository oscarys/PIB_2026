\begin{frame}{Clinical Applications II: Psychiatry \& Consciousness}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Psychiatric biomarkers}
      \begin{itemize}
        \item \textbf{Depression}: frontal $\alpha$ asymmetry (F4-F3)
          predicts treatment response \cite{Harmon-Jones2003}
        \item \textbf{Schizophrenia}: MMN amplitude reduction;
          $\gamma$ oscillation deficit
        \item \textbf{ADHD}: elevated $\theta/\beta$ ratio
          (controversial as standalone biomarker)
        \item \textbf{Alzheimer's}: slowing of peak $\alpha$ frequency,
          reduced coherence
      \end{itemize}
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{Disorders of consciousness (DOC)}
      \begin{itemize}
        \item \textbf{PCI} (perturbational complexity index, TMS-EEG):
          discriminates VS from MCS \cite{Casali2013}
        \item Covert awareness detection via P300 BCI
        \item Resting-state connectivity predicts prognosis
      \end{itemize}
      \vspace{0.3em}
      \textbf{Neurofeedback}
      \begin{itemize}
        \item Real-time SMR / $\theta$-$\beta$ training for ADHD
        \item SCP neurofeedback for ALS communication
        \item Closed-loop brain stimulation \cite{Sitaram2017}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
