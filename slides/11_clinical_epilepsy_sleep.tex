\begin{frame}{Aplicaciones clínicas I: epilepsia y sueño}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Epilepsia}
      \begin{itemize}
        \item Detección automática de crisis: CNN/LSTM en iEEG
          alcanza $>$95\% sensibilidad \cite{Subasi2005}
        \item Predicción de crisis: rasgos en el espacio de fase, 30--60 min
          ventana pre-ictal
        \item Espigas: filtrado por templete
      \end{itemize}
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{Análisis del sueño}
      \begin{itemize}
        \item Reglas de la AASM (W/N1/N2/N3/REM) automatizables con $\sim$85\% de precisión por época
        \item U-Sleep: modelo universal para diferentes montajes polisomnográficos \cite{Perslev2021}
      \end{itemize}
      \vspace{0.3em}
      \begin{block}{Herramientas clave}
        \small EEGLAB, MNE-Python, FieldTrip, Brainstorm
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
